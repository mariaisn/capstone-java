public class Test {

  public static void main(String[] args) {
    int[] numbers = {1,2,3,4,5};

    for(int index=0;index<5;index++)
    {
      System.out.print(numbers[index] + " ");
    }

    System.out.println();
    System.out.println(numbers.length);
  }
}
