public class FutureTuition {

	public static void main(String[] args) {
		double tuition = 10000.0;
		int year = 0;

		while (tuition < 20000.0) {
			tuition = tuition * 1.07;
			year++;
		}

		System.out.println("Tuition will be doubled in " + year + " years");
	}

}
